# CICD
Learning CICD setup

add user name: git config --global user.name "amarshoor"
add email: git config --global user.email "amarshoor@gmail.com"
create directory: mkdir CICDWorkspace
git initialized: git init
list files in directory: dir
see file not staged: git status
add file: git add
add all: 'git add --all' or 'get add .'
remove file: git rm --cached '<FileName>'
commit: git commit -m "<desc>"
add the remote: git remote add origin <URLRepositoryPath>
push the commited file to origin: git push -u origin
